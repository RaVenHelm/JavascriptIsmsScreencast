Javascript (ES5) Classes

                    
                        function Car(make, model, year, color) {
                            this.make = make;
                            this.model = model;
                            this.year = year;
                            this.color = color;
                        }

                        Car.prototype.start = function(delay) {
                            setTimeout(function() {
                                console.log(this.getFullCarName() + ' started!');
                            }.bind(this));
                        }

                        Car.prototype.getFullCarName() = function() {
                            return this.year + this.make + this.model;
                        }

                        Car.prototype.changeColor = function(newColor) {
                            this.color = newColor;
                        }

                        var myCar = new Car('Chevrolet', 'Corvette Stingray', 2015, 'Red');
                        myCar.start();
